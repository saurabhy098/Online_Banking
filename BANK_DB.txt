/*PERSONAL ACCOUNT DETAILS TABLE*/
  CREATE TABLE PERSONAL_ACCOUNT_DET(
  ACCOUNT_NO NUMBER(15) PRIMARY KEY NOT NULL,
  TITLE VARCHAR(4) NOT NULL,
  FIRST_NAME VARCHAR(10) NOT NULL,
  LAST_NAME VARCHAR(10) NOT NULL,
  FATHER_NAME VARCHAR(10) NOT NULL,
  MOBILE_NO NUMBER(10) NOT NULL,
  EMAIL_ID VARCHAR(20),
  AADHAR_NO NUMBER(12) NOT NULL,
  DOB DATE NOT NULL);

/*ADDRESS ACCOUNT DETAILS TABLE*/
  CREATE TABLE ADDRESS_ACCOUNT_DET(
  ACCOUNT_NO NUMBER(15) REFERENCES PERSONAL_ACCOUNT_DET(ACCOUNT_NO) NOT NULL,
  ADDRESS_LINE1 VARCHAR(20) NOT NULL,
  ADDRESS_LINE2 VARCHAR(20),
  LANDMARK VARCHAR(10) NOT NULL,
  USER_STATE VARCHAR(12) NOT NULL,
  CITY VARCHAR(15) NOT NULL,
  PINCODE NUMBER(6) NOT NULL);

/*OCCUPATION ACCOUNT DETAILS*/  
  CREATE TABLE OCCUPATION_ACCOUNT_DET(
  ACCOUNT_NO NUMBER(15) REFERENCES PERSONAL_ACCOUNT_DET(ACCOUNT_NO) NOT NULL,
  OCCUPATION_TYPE VARCHAR(10) NOT NULL,
  SOURCE_OF_INCOME VARCHAR(20) NOT NULL,
  GROSS_ANNUAL_INCOME NUMBER(7) NOT NULL);
  
/*REGISTER TABLE*/  
  CREATE TABLE USER_REGISTER(
  ACCOUNT_NO NUMBER(15) REFERENCES PERSONAL_ACCOUNT_DET(ACCOUNT_NO) NOT NULL,
  USER_ID VARCHAR(10) PRIMARY KEY NOT NULL,
  USER_PASSWORD VARCHAR(8) NOT NULL,
  TRANSACTION_PASSWORD NUMBER(4) NOT NULL);
  
/*LOGIN TABLE*/  
  CREATE TABLE USER_LOGIN(
  USER_ID VARCHAR(10) REFERENCES REGISTER(USER_ID) NOT NULL,
  USER_PASSWORD VARCHAR(8) NOT NULL);
  
/*PAYEE TABLE*/  
  CREATE TABLE PAYEE(
  NICK_NAME VARCHAR(10) NOT NULL,
  RECEIVER_NAME VARCHAR(10) NOT NULL,
  RECEIVER_ACCOUNT_NO NUMBER(15) PRIMARY KEY NOT NULL,
  SENDER_ACCOUNT_NO NUMBER(10) NOT NULL);
  
/*TRANSACTION TABLE*/  
  CREATE TABLE USER_TRANSACTION(
  TRANSACTION_DATE DATE NOT NULL,
  SENDER_ACCOUNT_NO NUMBER(15) REFERENCES PERSONAL_ACCOUNT_DET(ACCOUNT_NO) NOT NULL,
  RECEIVER_ACCOUNT_NO NUMBER(15) REFERENCES PAYEE(RECEIVER_ACCOUNT_NO) NOT NULL,
  TRANSACTION_TYPE VARCHAR(4)NOT NULL,
  CURRENT_BALANCE NUMBER(7) NOT NULL,
  AMOUNT_TRANSFERRED NUMBER(7) NOT NULL,
  FLAG NUMBER(1) NOT NULL);
  
/*CUSTOMER_BALANCE TABLE*/
  CREATE TABLE CUSTOMER_BALANCE(
  ACCOUNT_NO NUMBER(15) REFERENCES PERSONAL_ACCOUNT_DET(ACCOUNT_NO) NOT NULL,
  BALANCE NUMBER NOT NULL);
